#include <SFML/Graphics.hpp>
#include <GL/gl.h>
#include <unistd.h>
#include "vec2.hpp"
#include <cstdlib>
#include <time.h>
#include <vector>

#include <iostream>

using namespace sf;
using namespace std;

void glVertex2v(vec2 v)
{
    glVertex2f(v.x, v.y);
}

void drawVectorPoints(vector<vec2> *v)
{
    glBegin(GL_POINTS);
    for(int i=0; i<v->size(); i++)
        glVertex2v(v->at(i));
    glEnd();
}

void addPoints(vector<vec2> *initial_points, vector<vec2> *points, int add_points_count)
{
    for(int i=0; i<add_points_count; i++)
        {
            int rand_point = rand()%initial_points->size();

            vec2 new_point = (points->at(points->size()-1) + 2.0*initial_points->at(rand_point))/3.f;

            points->push_back(new_point);
        }
}

void addMiddlePoints(vector<vec2> *points)
{
    int points_count = points->size();

    for(int i=0; i<points_count-1; i++)
    {
        vec2 middle = (points->at(i) + points->at(i+1))/2.0;
        points->push_back(middle);
    }

    vec2 middle = (points->at(0) + points->at(points_count-1))/2.0;
    points->push_back(middle);
}

int main(int argc, char **argv)
{
    srand(time(NULL));

    vector<vec2> initial_points;

    float delay = 100;

    int add_points_count = 5;
    if(argc > 1)
        add_points_count = atoi(argv[1]);

    Clock delay_timer;

    vector<vec2> points;

    RenderWindow window(VideoMode(1920, 1080), "Chaos");

    bool is_points_input_end = false;
    bool mouse_left_pressed = false;
    bool space_pressed = false;

	Font font;
	font.loadFromFile("fonts/arial.ttf");
	Text text;
	text.setFont(font);
	text.setString("Hello World");
	text.setFillColor(Color::White);
	text.setOutlineColor(Color::White);
	text.setOutlineThickness(3.f);
	text.setCharacterSize(40);
	text.setPosition(50,50);

    while(window.isOpen())
    {
        Event event;
        while(window.pollEvent(event))
        {
            if(event.type == Event::Closed) window.close();
        }
        window.clear();
		window.draw(text);

        if(!is_points_input_end)
        {
            if(Mouse::isButtonPressed(Mouse::Left) && !mouse_left_pressed)
            {
                Vector2i mouse_coords = Mouse::getPosition(window);
                mouse_coords.y = window.getSize().y - mouse_coords.y;
                mouse_coords -= Vector2i(window.getSize())/2;

                vec2 mouse_pos(mouse_coords.x/float(window.getSize().x/2.0), mouse_coords.y/float(window.getSize().y/2.0));

                if((abs(mouse_pos.x) <= 1.0) && (abs(mouse_pos.y) <= 1.0))
                    initial_points.push_back(mouse_pos);

                mouse_left_pressed = true;
            }

            if(Keyboard::isKeyPressed(Keyboard::Space) && !space_pressed)
            {
                addMiddlePoints(&initial_points);

                space_pressed = true;
            }

            if(Keyboard::isKeyPressed(Keyboard::Return))
            {
                is_points_input_end = true;
            }

            if(mouse_left_pressed && !Mouse::isButtonPressed(Mouse::Left))
                mouse_left_pressed = false;

            if(space_pressed && !Keyboard::isKeyPressed(Keyboard::Space))
                space_pressed = false;

            drawVectorPoints(&initial_points);
        }
        else
        {
            if(delay_timer.getElapsedTime().asMicroseconds() > delay)
            {
                addPoints(&initial_points, &points, add_points_count);

                delay_timer.restart();
            }

            drawVectorPoints(&initial_points);
            drawVectorPoints(&points);
        }

        window.display();
        usleep(10000);

        if(Keyboard::isKeyPressed(Keyboard::Q)) window.close();
    }

    return 0;
}
